Golang入门篇|3.流程控制语句

Posted on May 3, 2022
tl;dr: Golang入门篇|3.流程控制语句

自增自减 ++ –

package main

import "fmt"

func main() {
	a := 0
	a++            // +1
	fmt.Println(a) // 1
	a--            // -1
	fmt.Println(a) // 0
}

条件语句 if else

func main() {
	if i, ok := xxx.xxx; ok { // 高级用法
		fmt.Println("当前为true")
	} else {
		fmt.Println("当前为false")
	}
}

选择语句 switch

func main() {
	a := 1
	switch a {
	case 0:
		fmt.Println("a = 0")
	case 1:
		fmt.Println("a = 1")
		fallthrough // 即使找到了匹配，也会继续执行到下一个

	case 3:
		fmt.Println("a = 3")
	default: // 上面都不成立
		fmt.Println("a == null")
	}
}

循环语句 for

// 最简单的for循环
	for {
		fmt.Println("我是循环")
		time.Sleep(2 * time.Second) // cpu休克两秒
	}
// 加条件的复杂循环
package main

import (
	"fmt"
	"time"
)

func main() {

	for i := 0; i < 10; i++ { // 1 < 10 = true执行，高级操作 i < 10 && i > 0
		fmt.Println(i)
		time.Sleep(time.Second) // cpu休克1秒
	}
}
