  • 类型推断，函数接收值手动设置类型
// 无法确定值时（类型推断），编译器会认为是 any类型，可接收任意值，会造成代码不健壮
function sum(a:number ,b: number){
    return a + b
}

console.log(sum(222,3))

// 指定函数返回类型
function sum(a:number ,b: number): string{
    return '结果：'+ a + b
}

console.log(sum(222,3))
  • 数组

// 数组为string和number 类型

a.push(23)

 let a = ['sss',2,true] // (string | number | boolean)[]

// 手动设置数组类型

  • unknown （我不知道什么类型，但有类型）//会进行类型监测
let b: unknown = '123'
console.log(b)

let c: string = b // 提示报错，c指定了类型，b也有类型，只是不知道什么类型

let c: string = b as string // 类型断言，告诉b的类型是string
console.log(c) // 123


// 已知类型和为止类型之间的转换
let a:number = 123

let b: string = a as unknown as string // 1.将number类型转成unknown类型 2.在讲unknown转换成string
console.log(b)

// 箭头函数
let msg = (): void => {
    console.log('ss')
}
msg()

let msg = (): void => console.log('ss') // 只有一行代码时可以简写

// type 函数参数
let msg = (user: { name: string, age: number }): void => {
    console.log('ss')
}
msg({ name: 'sss', age: 12 }) // ss

// 优化写法
type userType = {name: string, age: number}

let msg = (user: userType): void => {
    console.log('ss')
}
msg({ name: 'sss', age: 12 })

// 函数结构定义

}

// 优化
type userType = {name: string,age: number}
type userAddFun = (user: userType) => boolean

let addUser: userAddFun = (user:userType): boolean => {
    console.log('ss')
    return true
}

addUser({name:'ss',age:12})
