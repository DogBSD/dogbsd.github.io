  • 类型推断，函数接收值手动设置类型
// 无法确定值时（类型推断），编译器会认为是 any类型，可接收任意值，会造成代码不健壮
function sum(a:number ,b: number){
    return a + b
}

console.log(sum(222,3))

// 指定函数返回类型
function sum(a:number ,b: number): string{
    return '结果：'+ a + b
}

console.log(sum(222,3))
  • 数组
let a = ['sss'] //string[] 数组里的值只能是string
a.push('s')

console.log(a) // ==> [ 'sss', 's' ]


// 数组为string和number 类型
let a = ['sss',2] // (string | number)[]
a.push('s')
a.push(23)
a.push(23)

console.log(a) // ==> [ 'sss', 2, 's', 23, 23 ]

 let a = ['sss',2,true] // (string | number | boolean)[]

// 手动设置数组类型

let a:(number|boolean|string)[] = ["hang",23,false]
a.push("sss",12,true)
console.log(a) // ==> [ 'hang', 23, false, 'sss', 12, true ]
  • 对象的类型推断
const user = {name: 'hang',age: 23,open: true}
console.log(user.age) // 23

user.age = 34
console.log(user.age) // 34


// lesson是一个数组类型,数组的值是对象结构
const user = {name: 'hang',age: 23,open: true,lesson:[{title:"string"},{title:"number"}]}
console.log(user)

user.lesson.push({title:"ss"})
console.log(user)
console.log(user.lesson[1].title) // number

// 自动推断的类型
