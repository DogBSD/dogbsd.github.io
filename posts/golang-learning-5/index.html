Golang入门篇|5.Map

Posted on May 5, 2022
tl;dr: Golang入门篇|5.Map

三种声明方式

package main

import "fmt"

func main() {
	var m map[string]string // 简单声明
	m["bsd"] = "openbsd"

	m1 := map[string]string{}
	m1["bsd"] = "openbsd"
	m2 := make(map[string]string)
	m2["bsd"] = "openbsd"
	fmt.Println(m, m1, m2)
}

map 高级定义数据类型

package main

import "fmt"

func main() {

	m1 := map[int]interface{}{} //定义接受任意数据类型
	m1[1] = "openbsd"
	m1[2] = []int{1, 2, 3, 4}

	fmt.Println(m1[2])
}

简单操作数组

// 删除
	delete(m1, 1) // 删除map的key为1的值

// 循环
	for k, v := range m1 {
		fmt.Println(k, v)
	}
