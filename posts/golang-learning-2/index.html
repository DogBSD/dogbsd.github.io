package main

import "fmt"

func main() {
	var a string = "3.1415926"
	fmt.Printf("%T \n", a) // \n 为换行符，fmt包下的Printf格式化输出函数，%T 变量的类型
}

类型转换

// 一般可以转换的类型
float64 <==> int64 // 浮点转换为整型，小数点为直接丢失
int64 ==> string
string == > int64 // 不是所有的string都可以转换，但var a string = "3.1415926" 是可以转换成的


package main

import (
	"fmt"
	"log"
	"strconv"
)

func main() {
	var a string = "3"
	fmt.Printf("%T \n", a)
	int, err := strconv.Atoi(a) // 转换，返回两个值
	if err != nil {
		log.Fatal(err)
	}
	fmt.Println(int)
}


//  float64 <==> int64
package main

import (

func main() {
	var a float64 = 3.9999
	fmt.Println(int64(a)) // 3
