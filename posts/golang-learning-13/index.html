package main

import (
	"fmt"
	"log"
	"os"
)

func main() {
	f, err := os.OpenFile("./test.txt", os.O_CREATE|os.O_RDWR, 0777)
	if err != nil {
		log.Fatal(err)
		return
	}
	for {
		b := make([]byte, 12)
		n, err := f.Read(b)
		if err != nil {
			fmt.Println(err)
			return
		}
		fmt.Println(string(b), n)
	}
}
