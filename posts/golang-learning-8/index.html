Golang入门篇|8.结构体

Posted on May 6, 2022
tl;dr: Golang入门篇|8.结构体

结构体是一个可以储存不同数据类型的数据类型

声明

package main

import "fmt"

type Person struct {
	Name    string
	Age     int
	Hobbies []string
}

func main() {
	var p = Person{Name: "bsd", Age: 40, Hobbies: []string{"game", "eat"}}
	fmt.Println(p) // {bsd 40 [game eat]}
}

// or
	p1 := new(Person)
	p1.Age = 12
	fmt.Println(p1)

结构体方法

package main

import "fmt"

type Person struct {
	Name    string
	Age     int
	Hobbies []string
}

func (p Person) song(a string) (rest string) {
	fmt.Println(p)
	p.Name = a
	fmt.Println(p.Name)
	return p.Name
}
func main() {
	p1 := Person{}
	pp := p1.song("bsd")
	fmt.Println(pp)
}

结构体的嵌套

package main

import "fmt"

type Person struct {
	Name    string
	Age     int
	Hobbies []string
	Ss      Ss // 嵌套其他结构体
	Zs1 // 这个结构体，不是别名；匿名结构体
}

type Ss struct {
	S1 string
}
type Zs1 struct {
	H1 string
}


func main() {
	p := Person{}
	p.Ss.S1 = "hang"
	fmt.Println(p)
	fmt.Println(p.Ss)

	p.H1 = "sss"
	fmt.Println(p.H1) // 不存在歧义，可以直接调用。

}

子结构体挂在方法

package main

import "fmt"

type Person struct {
	Name    string
	Age     int
	Hobbies []string
	Ss      Ss // 嵌套其他结构体
	Zs1        // 这个结构体，不是别名；匿名结构体
}

type Ss struct {
	S1 string
}
type Zs1 struct {
	H1 string
}

func (s *Ss) Home() {
	fmt.Println("Open", s.S1)
}

func (s *Zs1) Home() {
	fmt.Println("Open", s.H1)
}

func main() {
	p := Person{}
	p.Ss.S1 = "hang"
	fmt.Println(p)
	fmt.Println(p.Ss)
	p.H1 = "sss"
	fmt.Println(p.H1) // 不存在歧义，可以直接调用。
	p.Ss.Home()
	p.Home()
}
