Golang入门篇|4。数组和切片

Posted on May 3, 2022
tl;dr: Golang入门篇|4。数组和切片

数组 array

把同类元素数组放入一个集合

// 定义一个数组
package main

import "fmt"

func main() {
	arr := [3]int{1, 2, 3} // 长度为3的，类型为int的数组
	fmt.Println("数组的长度:", len(arr))
	fmt.Println(arr)
}

arr := [...]int{1,2,3,4,5,6,7} // 固定长度，不一定需要指定


// 使用new的方式声明数组
var a = new([10]int)
a[5] = 3
fmt.Println(a)

  • 数组的缺点，长度不可变，意为这有新数据就无法添加进去，只能重新生成新的数组，消耗了内存。

  • 数组的循环

	for i := 0; i < len(a); i++ {
		fmt.Println(a[i])
	}

	for k, v := range a {
		fmt.Println(k, v)
	}
  • cap len
// 数组的长度和容量是相等的，因为数组不可以增加长度
if len(a) == cap(a) {
	fmt.Println("相等")
}
  • 二维数组
	arr := [3][3]int{ // 第一个三：三个元素；第二个三：长度为三位的int类型的数组
		{0, 1, 2}, // 第一个元素
		{2, 3, 4},
		{3, 4, 5},
	}

	fmt.Println(arr, "长度：", len(arr))
	fmt.Println(arr[0][2])

切片 slice

切片是数组的一部分

	a := [3]int{1, 2, 3} // 数组

	cl := a[1:] // 从数组上创建切片
	fmt.Println(cl)
	cl[0] = 5
	fmt.Println(cl)
	fmt.Println(a) // 数组的值也发生了变化，所以切片是引用类型

	// 切片的取值范围
	cl := a[:] // 整个数组
	cl := a[0:2] // 不包括第三个元素
	cl := a[:2] // 不包括第三个元素

	// 给切片增加一个值
	cl = append(cl, 4)



// 声明一个切片
var aa []int // 切片

a := make([]int, 4) // 可以给一个长度，增加会变
