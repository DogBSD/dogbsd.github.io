server


package main

import (
	"fmt"
	"log"
	"net"
)

func main() {
	tcpAddr, _ := net.ResolveTCPAddr("tcp", ":8080")
	listener, _ := net.ListenTCP("tcp", tcpAddr)

	for {
		Conn, err := listener.AcceptTCP()
		if err != nil {
			log.Fatal(err)
			return
		}
		go handleConnection(Conn)
	}
}

func handleConnection(conn *net.TCPConn) {
	for{
	buf := make([]byte, 1024)
	n,err := conn.Read(buf)
	if err!= nil{
		fmt.Println(err)
		break
	}
	fmt.Println(conn.RemoteAddr().String() + string(0:n))
	// fmt.Println(conn.RemoteAddr().String() + "连了一下")
	}
}

client

package main

import (
	"bufio"
	"net"
	"os"
)

func main() {
	tcpAddr, _ := net.ResolveTCPAddr("tcp", ":8080")
	conn, _ := net.DialTCP("tcp", nil, tcpAddr) // 建立一次tcp连接
	reader := bufio.NewReader(os.Stdin)
	bytes, _, _ := reader.ReadLine()
	conn.Write(bytes)
}
